// Search form handler - ONLY THIS FUNCTION IS MODIFIED (for image size) document.getElementById("searchForm").addEventListener("submit", async function (e) { e.preventDefault(); const keyword = document.getElementById("searchKeyword").value; const loadingEl = document.getElementById("searchLoading"); const resultsContainer = document.getElementById("searchResults"); console.log('Search form submitted with keyword:', keyword); loadingEl.style.display = "block"; resultsContainer.style.display = "none"; resultsContainer.innerHTML = ""; // Clear previous results try { console.log('Sending request to /api/search'); const response = await fetch("/api/search", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ keyword }), }); console.log('Response status from /api/search:', response.status); if (!response.ok) { const errorText = await response.text(); console.error('Search API error response:', errorText); let detail = errorText; try { const errJson = JSON.parse(errorText); detail = errJson.message || errorText; } catch (parseErr) { /* Do nothing, use raw errorText */ } throw new Error(`API request failed: ${detail}`); } const data = await response.json(); // data is the array from /api/search console.log('Search results data received from /api/search:', data); loadingEl.style.display = "none"; if (data && Array.isArray(data) && data.length > 0) { resultsContainer.style.display = "block"; data.forEach(result => { // result is one item: { question, answer, image } const resultElement = document.createElement("div"); resultElement.className = "search-result-item"; // Add Question const questionStrong = document.createElement('strong'); questionStrong.textContent = 'Q: '; resultElement.appendChild(questionStrong); resultElement.appendChild(document.createTextNode(result.question || '')); resultElement.appendChild(document.createElement('br')); // Add Answer const answerStrong = document.createElement('strong'); answerStrong.textContent = 'A: '; resultElement.appendChild(answerStrong); resultElement.appendChild(document.createTextNode(result.answer || '')); // --- START: Image Logic for Search Result Item --- if (result.image) { resultElement.appendChild(document.createElement('br')); const itemImageContainer = document.createElement('div'); itemImageContainer.style.marginTop = '8px'; const itemImgTag = document.createElement('img'); // --- MODIFIED SIZE HERE --- itemImgTag.style.maxWidth = '350px'; itemImgTag.style.maxHeight = '300px'; // Adjusted height // --- END MODIFIED SIZE --- itemImgTag.style.objectFit = 'contain'; itemImgTag.style.borderRadius = '8px'; itemImgTag.style.display = 'block'; const itemIframeTag = document.createElement('iframe'); // --- MODIFIED SIZE HERE --- itemIframeTag.style.width = '350px'; itemIframeTag.style.height = '250px'; // Adjusted height // --- END MODIFIED SIZE --- itemIframeTag.style.borderRadius = '8px'; itemIframeTag.style.border = 'none'; itemIframeTag.style.display = 'none'; const itemImgErrorMsg = document.createElement('p'); itemImgErrorMsg.style.color = '#ef4444'; itemImgErrorMsg.style.fontSize = '0.8rem'; itemImgErrorMsg.textContent = 'Unable to load image.'; itemImgErrorMsg.style.display = 'none'; let originalImageUrlForItem = ""; if (result.image && typeof result.image === 'string') { originalImageUrlForItem = result.image.trim(); } let attemptImageUrlForItem = originalImageUrlForItem; if (originalImageUrlForItem && originalImageUrlForItem.includes('drive.google.com')) { const fileIdMatch = originalImageUrlForItem.match(/[-\w]{25,}/); if (fileIdMatch && fileIdMatch[0]) { attemptImageUrlForItem = `https://drive.google.com/uc?export=download&id=${fileIdMatch[0]}`; } } if (attemptImageUrlForItem) { itemImgTag.src = attemptImageUrlForItem; itemImgTag.onload = () => { itemImgTag.style.display = 'block'; itemIframeTag.style.display = 'none'; itemImgErrorMsg.style.display = 'none'; }; itemImgTag.onerror = () => { itemImgTag.style.display = 'none'; if (originalImageUrlForItem && originalImageUrlForItem.includes('drive.google.com')) { const fileIdMatch = originalImageUrlForItem.match(/[-\w]{25,}/); if (fileIdMatch && fileIdMatch[0]) { const iframeUrl = `https://drive.google.com/file/d/${fileIdMatch[0]}/preview`; itemIframeTag.src = iframeUrl; itemIframeTag.style.display = 'block'; itemImgErrorMsg.style.display = 'none'; } else { itemIframeTag.style.display = 'none'; itemImgErrorMsg.style.display = 'block'; } } else if (originalImageUrlForItem) { itemIframeTag.style.display = 'none'; itemImgErrorMsg.style.display = 'block'; } else { itemImgErrorMsg.style.display = 'block'; } }; } else { itemImgErrorMsg.style.display = 'block'; } itemImageContainer.appendChild(itemImgTag); itemImageContainer.appendChild(itemIframeTag); itemImageContainer.appendChild(itemImgErrorMsg); resultElement.appendChild(itemImageContainer); } // --- END: Image Logic for Search Result Item --- resultsContainer.appendChild(resultElement); }); } else if (data && Array.isArray(data) && data.length === 0) { resultsContainer.style.display = "block"; resultsContainer.innerHTML = '
No results found.
'; } else { console.error('Search results: Data received from API is not a valid array or is undefined/null', data); resultsContainer.style.display = "block"; resultsContainer.innerHTML = '
Error: Could not properly load search results.
'; } } catch (error) { console.error('Search error in frontend:', error); loadingEl.style.display = "none"; resultsContainer.style.display = "block"; resultsContainer.innerHTML = `
Error searching. ${error.message || 'Please try again later.'}
`; } });
