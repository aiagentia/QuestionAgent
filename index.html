// New search form handler (with more robust image display logic) document.getElementById("searchForm").addEventListener("submit", async function (e) { e.preventDefault(); const keyword = document.getElementById("searchKeyword").value; const loadingEl = document.getElementById("searchLoading"); const resultsContainer = document.getElementById("searchResults"); console.log('Search form submitted with keyword:', keyword); loadingEl.style.display = "block"; resultsContainer.style.display = "none"; resultsContainer.innerHTML = ""; // Clear previous results try { console.log('Sending request to /api/search'); const response = await fetch("/api/search", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ keyword }), }); console.log('Response status from /api/search:', response.status); if (!response.ok) { const errorText = await response.text(); console.error('Search API error response:', errorText); let detail = errorText; try { const errJson = JSON.parse(errorText); detail = errJson.message || errorText; } catch (parseErr) { /* Do nothing, use raw errorText */ } throw new Error(`API request failed: ${detail}`); } const data = await response.json(); // data is the array from /api/search console.log('Search results data received from /api/search:', data); loadingEl.style.display = "none"; if (data && Array.isArray(data) && data.length > 0) { resultsContainer.style.display = "block"; data.forEach(result => { // result is one item: { question, answer, image } const resultElement = document.createElement("div"); resultElement.className = "search-result-item"; // Add Question const questionStrong = document.createElement('strong'); questionStrong.textContent = 'Q: '; resultElement.appendChild(questionStrong); resultElement.appendChild(document.createTextNode(result.question || '')); resultElement.appendChild(document.createElement('br')); // Add Answer const answerStrong = document.createElement('strong'); answerStrong.textContent = 'A: '; resultElement.appendChild(answerStrong); resultElement.appendChild(document.createTextNode(result.answer || '')); // Add Image (if it exists) with robust Google Drive handling if (result.image) { resultElement.appendChild(document.createElement('br')); // Space before image const itemImageContainer = document.createElement('div'); itemImageContainer.style.marginTop = '8px'; const itemImgTag = document.createElement('img'); itemImgTag.style.maxWidth = '200px'; // itemImgTag.style.maxHeight = '200px'; // Optional: constrain height // itemImgTag.style.objectFit = 'contain'; // Optional: how image fits itemImgTag.style.borderRadius = '8px'; itemImgTag.style.display = 'block'; // Start by trying to display the img tag const itemIframeTag = document.createElement('iframe'); itemIframeTag.style.width = '200px'; itemIframeTag.style.height = '150px'; // Adjust as needed itemIframeTag.style.borderRadius = '8px'; itemIframeTag.style.border = 'none'; itemIframeTag.style.display = 'none'; // Hide iframe initially const itemImgErrorMsg = document.createElement('p'); itemImgErrorMsg.style.color = '#ef4444'; // Red color for error itemImgErrorMsg.style.fontSize = '0.8rem'; itemImgErrorMsg.textContent = 'Unable to load image.'; itemImgErrorMsg.style.display = 'none'; // Hide error initially let originalImageUrl = ""; if (result.image && typeof result.image === 'string') { originalImageUrl = result.image.trim(); } let attemptImageUrl = originalImageUrl; if (originalImageUrl && originalImageUrl.includes('drive.google.com')) { const fileIdMatch = originalImageUrl.match(/[-\w]{25,}/); if (fileIdMatch && fileIdMatch[0]) { attemptImageUrl = `https://drive.google.com/uc?export=download&id=${fileIdMatch[0]}`; console.log(`Search Item: GDrive direct download URL: ${attemptImageUrl}`); } else { console.log(`Search Item: GDrive URL, but no File ID found: ${originalImageUrl}`); // attemptImageUrl remains originalImageUrl, might still work or fail gracefully } } if (attemptImageUrl) { itemImgTag.src = attemptImageUrl; itemImgTag.onload = () => { console.log('Search Item: Image loaded successfully via :', itemImgTag.src); itemImgTag.style.display = 'block'; itemIframeTag.style.display = 'none'; itemImgErrorMsg.style.display = 'none'; }; itemImgTag.onerror = () => { console.error('Search Item: failed for:', attemptImageUrl, '. Trying iframe for original:', originalImageUrl); itemImgTag.style.display = 'none'; if (originalImageUrl && originalImageUrl.includes('drive.google.com')) { const fileIdMatch = originalImageUrl.match(/[-\w]{25,}/); if (fileIdMatch && fileIdMatch[0]) { const iframeUrl = `https://drive.google.com/file/d/${fileIdMatch[0]}/preview`; console.log(`Search Item: GDrive iframe URL: ${iframeUrl}`); itemIframeTag.src = iframeUrl; itemIframeTag.style.display = 'block'; itemImgErrorMsg.style.display = 'none'; } else { console.error('Search Item: GDrive URL but no File ID for iframe from:', originalImageUrl); itemIframeTag.style.display = 'none'; itemImgErrorMsg.style.display = 'block'; } } else { console.error('Search Item: Non-GDrive image failed, no iframe fallback for:', originalImageUrl); itemIframeTag.style.display = 'none'; itemImgErrorMsg.style.display = 'block'; } }; } else { // No valid image URL to attempt itemImgErrorMsg.style.display = 'block'; } itemImageContainer.appendChild(itemImgTag); itemImageContainer.appendChild(itemIframeTag); itemImageContainer.appendChild(itemImgErrorMsg); resultElement.appendChild(itemImageContainer); } resultsContainer.appendChild(resultElement); }); } else if (data && Array.isArray(data) && data.length === 0) { resultsContainer.style.display = "block"; resultsContainer.innerHTML = '
No results found.
'; } else { console.error('Search results: Data received from API is not a valid array or is undefined/null', data); resultsContainer.style.display = "block"; resultsContainer.innerHTML = '
Error: Could not properly load search results.
'; } } catch (error) { console.error('Search error in frontend:', error); loadingEl.style.display = "none"; resultsContainer.style.display = "block"; resultsContainer.innerHTML = `
Error searching. ${error.message || 'Please try again later.'}
`; } });
